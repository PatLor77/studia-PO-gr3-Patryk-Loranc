package com.company.Instruments
import java.time.LocalDate;

public abstract class Instrument {
    private String producent;
    private LocalDate rokProdukcji;

    public Instrument(String producent, LocalDate rokProdukcji) {
        this.producent = producent;
        this.rokProdukcji = rokProdukcji;
    }

    public String getProducent() {
        return this.producent;
    }

    public LocalDate getRokProdukcji() {
        return this.rokProdukcji;
    }

    public abstract void dzwiek();

    public boolean equals(String producent, LocalDate rokProdukcji) {
        return this.producent.equals(producent) && this.rokProdukcji.equals(rokProdukcji);
    }

    public String toString() {
        String var10000 = this.producent;
        return "Producent: " + var10000 + "\nRok produkcji: " + this.rokProdukcji.toString();
    }
}
